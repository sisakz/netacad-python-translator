<p>This is the resulting code:</p>

<codep class="copy">def is_a_triangle(a, b, c):
    return a + b > c and b + c > a and c + a > b


def heron(a, b, c):
    p = (a + b + c) / 2
    return (p * (p - a) * (p - b) * (p - c)) ** 0.5


<mark style="background-color:#e6f2ff;">def area_of_triangle(a, b, c):</mark>
    if not is_a_triangle(a, b, c):
        return None
    return heron(a, b, c)


print(area_of_triangle(1., 1., 2. ** .5))

</codep>
